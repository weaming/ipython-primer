def show_avg_info(avg):
    print "{:.1f}".format(avg.ix["PTS"])
    print "{:.1f}".format(avg.ix["REB"])
    print "{:.1f}".format(avg.ix["AST"])
    print "{:.1f}".format(avg.ix["BLK"])
    print "{:.1f}".format(avg.ix["MIN"])
    print "{:.1f}".format(avg.ix["STL"])
    print "{:.1f}".format(avg.ix["TOV"])
    print "{:.1f}".format(avg.ix["PF"])
    print "{:.1f}%".format(avg.ix["FGM"] * 100 / avg.ix["FGA"])
    print "{:.1f}%".format(avg.ix["FG3M"] * 100 / avg.ix["FG3A"])
    print "{:.1f}%".format(avg.ix["FTM"] * 100 / avg.ix["FTA"])
    print "{:.1f}".format(avg.ix["DREB"])
    print "{:.1f}".format(avg.ix["OREB"])
    print "{:.1f}".format(avg.ix["PLUS_MINUS"])
